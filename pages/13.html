Встроенные методы строки.

.toUpperCase()

К верхнему регистру.

  
    let str = 'Строка';

    console.log(str.toUpperCase()); //СТРОКА

.toLowerCase()

К нижнему регистру.

  
    let str = 'Строка';

    console.log(str.toLowerCase()); //строка

.length

Длинна строки

  
    let str = 'text';

    console.log(str.length); //4

.indexOf()

Поиск по символу или сочетанию (без учёта регистра). Выдаёт порядковый индекс. Если вывод -1, значит символ отсутствует.

  
    let str = 'text';

    console.log(str.indexOf('x')); //2

.includes()

Поиск по символу или сочетанию (без учёта регистра). Выдаёт true/false.

  
    let str = 'text';

    console.log(str.includes('x')); //true

Пример с поиском по товарам.

  
    const products = [
      'Яйцо столовое',
      'Молоко Молочник',
      'Коровка Молоко',
      'Виноград Изабелла',
    ];

    let search = 'молоко';

    products.forEach((product) => {
      if (product.toLowerCase().includes(search.toLowerCase())) {
        console.log(product);
      }
    });

    //Молоко Молочник
    //Коровка Молоко

.slice()

Извлечение части строки в виде новой, без изменения изначальной строки.

  
    let str = 'London is a capital of Great Britan';

    console.log(str.slice(1)); //ondon is a capital of Great Britan

Cрезает всё до индекса в скобках

  
    let str = 'London is a capital of Great Britan';

    console.log(str.slice(5)); //n is a capital of Great Britan

1й параметр - первый включённый порядковый номер символа. 2й параметр - это первый не включённый порядковый номер символа.

  
    let str = 'London is a capital of Great Britan';

    console.log(str.slice(0, 6)); //London

Пример с форматированием даты

  
    let dateStr = '2023-10-05';
    let year = dateStr.slice(0, 4); // Извлекаем год
    let month = dateStr.slice(5, 7); // Извлекаем месяц
    let day = dateStr.slice(8, 10); // Извлекаем день
    console.log(year, month, day); // 2023 10 05

Можно так же срезать последний символ (или несколько, если использовать -2, -3 итд).

  
    let str = 'Hello World!';
    let newStr = str.slice(0, -1); //Удаляем последний символ
    console.log(newStr); //Hello World

.split()

Разбивает строки на массив

  
    let str = 'London is a capital of Great Britan';

    console.log(str.split(' ')); // ['London', 'is', 'a', 'capital', 'of', 'Great', 'Britan']

.join()

Объединяет все элементы массива или массивоподобных объектов в строку

  
    let str = 'London is a capital of Great Britan';

    console.log(str.split(' ').join(' ')); // London is a capital of Great Britan
  

  
    let str = 'London is a capital of Great Britan';

    console.log(str.split(' ').join('|')); // London|is|a|capital|of|Great|Britan

.trim()

Убирает все пробелы в начале и конце строки (полезен для работы с формами к примеру)

  
    let str = '  London is a capital of Great Britan ';

    console.log(str.trim()); //London is a capital of Great Britan
